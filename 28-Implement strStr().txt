class Solution {
    public int strStr(String haystack, String needle) {   
        if(needle.isEmpty()){
            return 0;
        }else{
            return haystack.indexOf(needle);
        }
    }
}
	
	
	class Solution {
public:
    int strStr(string haystack, string needle) {
        if(needle.length() == 0){
            return 0;
        }else{
            return haystack.find(needle);
        }
    }
};
